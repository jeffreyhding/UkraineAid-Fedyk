# for use with fpds.gov data, which does not come with fiscal year column
calculate_fiscal_year <- function(date) {
  fiscal_year <- if_else(
    month(date) >= 10, 
    year(date) + 1, 
    year(date) 
  )
  return(fiscal_year)
}
